```cpp
LRESULT Window::HandleMessage(HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam) {
    switch (uMsg) {
        case WM_CREATE:
            OnCreate();
            return 0;

        case WM_COMMAND:
            OnCommand(wParam);
            return 0;

        case WM_PAINT:
            OnPaint();
            return 0;

        case WM_SIZE:
            OnSize(LOWORD(lParam), HIWORD(lParam));
            return 0;

        case WM_TIMER:
            OnTimer(wParam);
            return 0;

        case WM_DESTROY:
            OnDestroy();
            PostQuitMessage(0);
            return 0;

        default:
            return DefWindowProc(hwnd, uMsg, wParam, lParam);
    }
}
```

```cpp
const int STAGE_PROTOPLASM = 0;          // 原生质阶段
const int STAGE_ORGANELLES_START = 1;    // 细胞器阶段开始
const int STAGE_ORGANELLES_END = 10;     // 细胞器阶段结束
const int STAGE_MULTICELL_START = 11;    // 多细胞阶段开始
const int STAGE_MULTICELL_END = 30;      // 多细胞阶段结束
const int STAGE_SENTIENCE_START = 31;    // 感知阶段开始

const int STAGE_CIVILIZATION = 61;       // 文明阶段
```

### 状态转换逻辑

```cpp
void Window::ProcessEvolution() {
    // 检查是否达到新的进化阶段
    int newStage = CalculateEvolutionStage();

    if (newStage > evolutionStage_) {
        evolutionStage_ = newStage;
        OnEvolutionStageChanged();
    }

    // 更新按钮可见性
    UpdateButtonVisibility();

    // 更新进度显示
    UpdateEvolutionProgress();
}

int Window::CalculateEvolutionStage() {
    int stage = STAGE_PROTOPLASM;

    // 基于DNA数量计算基础阶段
    if (dnaCount_ >= 100) stage = STAGE_ORGANELLES_START;
    if (dnaCount_ >= 1000) stage = STAGE_ORGANELLES_END;
    if (dnaCount_ >= 10000) stage = STAGE_MULTICELL_START;
    if (dnaCount_ >= 100000) stage = STAGE_MULTICELL_END;
    if (dnaCount_ >= 1000000) stage = STAGE_SENTIENCE_START;
    if (dnaCount_ >= 10000000) stage = STAGE_CIVILIZATION;

    // 根据解锁的结构调整阶段
    if (nucleusCount_ > 0) stage = max(stage, STAGE_MULTICELL_START);
    if (mitochondriaCount_ > 5) stage = max(stage, STAGE_ORGANELLES_END);

    return stage;
}
```

```cpp
void Window::ResizeControls(int width, int height) {
    // === 计算布局参数 ===
    const int padding = 20;          // 边距
    const int menuH = 40;            // 菜单高度
    const int btnGap = 10;           // 按钮间距

    // 三列布局
    int leftW = 200;                 // 左列宽度
    int rightW = 200;                // 右列宽度
    int centerW = width - leftW - rightW - (padding * 3);  // 中央列宽度

    // === 左侧面板布局 ===
    int leftX = padding;
    int contentY = 80;

    MoveWindow(hLabelRNA_, leftX, contentY, leftW, 30, TRUE);
    MoveWindow(hBtnRNA_, leftX, contentY + 30, leftW, menuH, TRUE);
    MoveWindow(hLabelDNA_, leftX, contentY + 80, leftW, 30, TRUE);
    MoveWindow(hBtnDNA_, leftX, contentY + 110, leftW, menuH, TRUE);

    // === 中央区域动态布局 ===
    int centerX = leftW + padding * 2;
    int gridY = 100;
    int btnW = (centerW - btnGap * 3) / 4;  // 4列布局
    int btnH = 50;

    // 根据进化阶段显示不同按钮
    if (evolutionStage_ >= STAGE_ORGANELLES_START) {
        // 第一行：基础细胞器
        MoveWindow(hBtnMitochondria_, centerX, gridY, btnW, btnH, TRUE);
        MoveWindow(hBtnChloroplast_, centerX + btnW + btnGap, gridY, btnW, btnH, TRUE);
        MoveWindow(hBtnNucleus_, centerX + (btnW + btnGap) * 2, gridY, btnW, btnH, TRUE);
        MoveWindow(hBtnRibosome_, centerX + (btnW + btnGap) * 3, gridY, btnW, btnH, TRUE);
    }

    // === 右侧菜单布局 ===
    int rightX = width - rightW - padding;
    int menuY = 80;
    int menuGap = 10;

    MoveWindow(hBtnSettings_, rightX, menuY, rightW, menuH, TRUE);
    MoveWindow(hBtnProgress_, rightX, menuY + menuH + menuGap, rightW, menuH, TRUE);
    MoveWindow(hBtnReset_, rightX, menuY + (menuH + menuGap) * 2, rightW, menuH, TRUE);
}
```

### 按钮可见性状态机

```cpp
void Window::UpdateButtonVisibility() {
    // === 获取当前游戏阶段 ===
    bool isPreReproduction = (evolutionStage_ < STAGE_MULTICELL_START);
    bool isReproductionPhase = (evolutionStage_ >= STAGE_MULTICELL_START && evolutionStage_ < STAGE_MULTICELL_END);
    bool isSentiencePhase = (evolutionStage_ >= STAGE_SENTIENCE_START && evolutionStage_ < STAGE_SENTIENCE_END);
    bool isCivilizationPhase = (evolutionStage_ >= STAGE_CIVILIZATION);

    // === 1. 繁殖前期阶段 ===
    if (isPreReproduction) {
        // 显示基础细胞器按钮
        ShowWindow(hBtnMitochondria_, SW_SHOW);
        ShowWindow(hBtnChloroplast_, SW_SHOW);
        ShowWindow(hBtnNucleus_, SW_SHOW);
        ShowWindow(hBtnRibosome_, SW_SHOW);

        // 隐藏高级选项
        ShowWindow(hBtnGovernment_, SW_HIDE);
        ShowWindow(hBtnAnarchy_, SW_HIDE);
    }

    // === 2. 繁殖选择阶段 ===
    if (isReproductionPhase) {
        // 隐藏基础细胞器
        ShowWindow(hBtnMitochondria_, SW_HIDE);
        ShowWindow(hBtnChloroplast_, SW_HIDE);

        // 显示繁殖方式选择
        ShowWindow(hBtnSexual_, SW_SHOW);
        ShowWindow(hBtnAsexual_, SW_SHOW);

        // 显示高级结构
        if (evolutionStage_ >= STAGE_MULTICELL_START + 5) {
            ShowWindow(hBtnMembrane_, SW_SHOW);
            ShowWindow(hBtnCellWall_, SW_SHOW);
        }
    }

    // === 3. 感知阶段 ===
    if (isSentiencePhase) {
        // 根据繁殖方式显示不同选项
        if (isSexualReproduction_) {
            // 显示9个种群按钮（3x3网格）
            for (int i = 0; i < 9; i++) {
                ShowWindow(hBtnSpecies_[i], SW_SHOW);
            }
        } else {
            // 无性繁殖显示变异选项
            ShowWindow(hBtnMutation1_, SW_SHOW);
            ShowWindow(hBtnMutation2_, SW_SHOW);
            ShowWindow(hBtnMutation3_, SW_SHOW);
        }

        // 显示政府形态选择
        ShowWindow(hBtnGovernment_, SW_SHOW);
        ShowWindow(hBtnAnarchy_, SW_SHOW);
    }

    // === 4. 文明阶段（游戏结束） ===
    if (isCivilizationPhase) {
        // 隐藏所有操作按钮
        HideAllGameButtons();

        // 显示游戏结束信息
        isGameEnded_ = true;
        InvalidateRect(hwnd_, nullptr, TRUE);
    }
}
```

### 基础进化计算

```cpp
void Window::OnRNAButtonClick() {
    // 基础RNA生成
    int baseGen = 10;

    // 细胞器加成
    if (chloroplastCount_ > 0) {
        baseGen += chloroplastCount_ * 5;  // 叶绿体提供额外RNA
    }

    if (ribosomeCount_ > 0) {
        baseGen *= (1 + ribosomeCount_ * 0.1);  // 核糖体提升效率
    }

    // 消耗计算
    int consumption = 5;
    if (nucleusCount_ > 0) {
        consumption += nucleusCount_ * 2;  // 细胞核增加消耗
    }

    // 应用生成
    rnaCount_ += baseGen;

    // 更新显示
    UpdateResourceDisplay();
    AddLog(L"生成 RNA +%d", baseGen);
}

void Window::OnDNAButtonClick() {
    // 需要消耗RNA生成DNA
    int rnaCost = 100;
    int baseGen = 1;

    // 细胞核加成
    if (nucleusCount_ > 0) {
        baseGen += nucleusCount_;
    }

    // 检查资源
    if (rnaCount_ >= rnaCost) {
        rnaCount_ -= rnaCost;
        dnaCount_ += baseGen;
        UpdateResourceDisplay();
        AddLog(L"生成 DNA +%d", baseGen);

        // 触发进化检查
        ProcessEvolution();
    } else {
        ShowCustomMessageBox(L"RNA不足", L"需要100 RNA才能生成DNA");
    }
}
```

```cpp
void Window::OnMitochondriaButtonClick() {
    int cost = 50;  // DNA成本

    if (dnaCount_ >= cost) {
        dnaCount_ -= cost;
        mitochondriaCount_++;

        // 更新RNA被动生成速率
        rnaRate_ += 2;  // 每个线粒体提供2点/秒的RNA

        UpdateResourceDisplay();
        AddLog(L"解锁线粒体 +%d", mitochondriaCount_);
        ProcessEvolution();
    }
}

void Window::OnNucleusButtonClick() {
    int cost = 500;  // 昂贵的DNA成本

    if (dnaCount_ >= cost && nucleusCount_ == 0) {  // 细胞核只能有一个
        dnaCount_ -= cost;
        nucleusCount_ = 1;

        // 细胞核提供多种加成
        rnaRate_ += 5;        // 提升RNA生成
        dnaRate_ += 0.5;      // 开始被动生成DNA
        evolutionStage_ = max(evolutionStage_, STAGE_MULTICELL_START);  // 直接进入多细胞阶段

        UpdateResourceDisplay();
        AddLog(L"进化出细胞核！进入多细胞阶段");
        ProcessEvolution();
    }
}
```

## 资源生成系统

### 被动生成算法

```cpp
void Window::ProcessPassiveGeneration() {
    if (!isGameStarted_ || isGameEnded_) return;

    // === 1. 计算基础速率 ===
    double rnaGeneration = 0;
    double dnaGeneration = 0;

    // === 2. 应用细胞器生成效果 ===
    // 线粒体：基础RNA生成
    rnaGeneration += mitochondriaCount_ * 2.0;

    // 叶绿体：光合作用生成RNA（需要光照条件）
    if (/* 有光照条件 */ true) {
        rnaGeneration += chloroplastCount_ * 1.5;
    }

    // 核糖体：提升RNA生成效率
    if (ribosomeCount_ > 0) {
        rnaGeneration *= (1.0 + ribosomeCount_ * 0.1);
    }

    // === 3. 应用细胞核效果 ===
    if (nucleusCount_ > 0) {
        // 细胞核提供基础DNA生成
        dnaGeneration += 0.5;

        // DNA复制加速
        dnaGeneration += evolutionStage_ * 0.1;

        // RNA到DNA的转化
        double rnaToDnaConversion = rnaCount_ * 0.01;  // 1%转化率
        if (rnaCount_ > 100) {
            rnaGeneration -= rnaToDnaConversion;
            dnaGeneration += rnaToDnaConversion * 0.1;
        }
    }

    // === 4. 应用阶段修正 ===
    if (evolutionStage_ >= STAGE_MULTICELL_START) {
        // 多细胞生物效率提升
        double efficiency = 1.0 + (evolutionStage_ - STAGE_MULTICELL_START) * 0.02;
        rnaGeneration *= efficiency;
        dnaGeneration *= efficiency;
    }

    // === 5. 应用种群效果 ===
    if (isSexualReproduction_ && evolutionStage_ >= STAGE_SENTIENCE_START) {
        // 有性繁殖的优势
        dnaGeneration *= 1.5;
    }

    // === 6. 更新资源 ===
    rnaCount_ += (int)rnaGeneration;
    dnaCount_ += (int)dnaGeneration;

    // === 7. 更新显示 ===
    if ((int)rnaGeneration > 0 || (int)dnaGeneration > 0) {
        UpdateResourceDisplay();

        if ((int)rnaGeneration > 0) {
            AddLog(L"被动生成 RNA +%d/s", (int)rnaGeneration);
        }
        if ((int)dnaGeneration > 0) {
            AddLog(L"被动生成 DNA +%d/s", (int)dnaGeneration);
        }
    }

    // === 8. 检查进化 ===
    ProcessEvolution();
}
```

### 消耗计算系统

```cpp
void Window::CalculateConsumption() {
    int rnaConsumption = 0;
    int dnaConsumption = 0;

    // 基础维持消耗
    if (evolutionStage_ > STAGE_PROTOPLASM) {
        rnaConsumption += 1;  // 基础代谢
    }

    // 细胞器维持消耗
    rnaConsumption += mitochondriaCount_ * 1;
    rnaConsumption += chloroplastCount_ * 0.5;
    rnaConsumption += nucleusCount_ * 5;
    rnaConsumption += ribosomeCount_ * 0.5;

    // 多细胞维持
    if (evolutionStage_ >= STAGE_MULTICELL_START) {
        int cellCount = evolutionStage_ - STAGE_MULTICELL_START + 1;
        rnaConsumption += cellCount * 2;
    }

    // 复杂结构消耗
    if (evolutionStage_ >= STAGE_SENTIENCE_START) {
        dnaConsumption += 1;  // 神经系统维持
    }
```

```cpp
void Window::DrawEvolutionProgressBar(HDC hdc, int width, int height) {
    const int barHeight = 30;
    const int barY = height - barHeight - 10;
    const int padding = 220;  // 左右面板宽度 + 边距
    const int barX = padding;
    const int barWidth = width - padding * 2;

    // === 背景轨道 ===
    RECT barRect = {barX, barY, barX + barWidth, barY + barHeight};
    HBRUSH hBrushBg = CreateSolidBrush(RGB(20, 20, 20));
    HPEN hPenBorder = CreatePen(PS_SOLID, 2, RGB(60, 60, 60));

    SelectObject(hdc, hBrushBg);
    SelectObject(hdc, hPenBorder);
    Rectangle(hdc, barRect.left, barRect.top, barRect.right, barRect.bottom);

    // === 计算进度 ===
    double progress = CalculateEvolutionProgress();
    int filledWidth = (int)(barWidth * progress);

    // === 绘制进度段 ===
    if (filledWidth > 0) {
        RECT fillRect = {barX, barY, barX + filledWidth, barY + barHeight};

        // 根据阶段选择颜色
        COLORREF barColor;
        if (evolutionStage_ < STAGE_MULTICELL_START) {
            barColor = RGB(100, 200, 100);  // 绿色 - 原生质
        } else if (evolutionStage_ < STAGE_SENTIENCE_START) {
            barColor = RGB(100, 150, 200);  // 蓝色 - 多细胞
        } else if (evolutionStage_ < STAGE_CIVILIZATION) {
            barColor = RGB(200, 200, 100);  // 黄色 - 感知
        } else {
            barColor = RGB(200, 100, 100);  // 红色 - 文明
        }

        // 创建渐变效果
        HBRUSH hBrushFill = CreateSolidBrush(barColor);
        SelectObject(hdc, hBrushFill);
        FillRect(hdc, &fillRect, hBrushFill);
        DeleteObject(hBrushFill);
    }

    // === 绘制阶段标记 ===
    DrawStageMarkers(hdc, barX, barY, barWidth, barHeight);

    // === 绘制进度文字 ===
    DrawProgressText(hdc, barX, barY, barWidth, barHeight);

    // 清理
    DeleteObject(hBrushBg);
    DeleteObject(hPenBorder);
}

double Window::CalculateEvolutionProgress() {
    if (evolutionStage_ >= STAGE_CIVILIZATION) {
        return 1.0;  // 100%
    }

    // 计算当前阶段内的进度
    int stageStart = GetStageStart(evolutionStage_);
    int stageEnd = GetStageEnd(evolutionStage_);
    int stageRange = stageEnd - stageStart;

    if (stageRange <= 0) return 0.0;

    int progressInStage = dnaCount_ - stageStart;
    double stageProgress = (double)progressInStage / stageRange;

    // 计算总体进度
    double totalStages = STAGE_CIVILIZATION - STAGE_PROTOPLASM;
    double completedStages = evolutionStage_ - STAGE_PROTOPLASM;

    return (completedStages + stageProgress) / totalStages;
}
```
